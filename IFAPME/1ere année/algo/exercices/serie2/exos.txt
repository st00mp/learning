### Exercice 7

# IN
nbrOuvriers (-> n'est pas une "variable")
nom
nbrHeures
txHorraire
anciennete
lieu
nbrEnfants
___
Par ouvrier * 100

# OUT
nom
salaire
prime
___
Par ouvrier * 100 -> totalAPayer

# TO CREATE 
salaire
prime
totalAPayer
__

totalAPayer = 0

For (i=1; 100; i++)
    Obtenir nom, nbrHeures, txHorraire, anciennete, lieu, nbrEnfants
    prime = 0
    primeEnfant = 0

    salaire = nbrHeures * txHorraire
    if nbrHeures > 36
        salaire += (nbrHeures -36) * txHorraire * 0,4 
    end if

    if anciennete > 6
        prime += 6*8 + (12 * (anciennete-6))
    else
        prime += (8 * anciennete)
    end if

    if lieu == 2
        prime += 250
    if lieu == 1
        prime += 125
    if lieu == 0
        prime += 0

    if nbrEnfants > 3
        primeEnfant += 3*20 + (24 * (nbrEnfants - 3))
    else
        primeEnfant += (20 * nbrEnfants)
    end if

    if primeEnfant > 140
        primeEnfant = 140

    prime += primeEnfant
    
    Afficher nom, salaire, prime   

    totalAPayer += (salaire + prime)

Afficher totalAPayer
